def execute
  user = User::User.new
  user.email = email
  user.password = password
  user.confirmed_at = nil
  user.save
  token = User::Token.new(user, User::Token::TYPE_CONFIRMATION)
  token.save
  Mailer.confirmation_email(user.email, token.code).deliver_later
  { id: user['_id'] }
end
