def if_the_credentials_are_right
  if User::User.where(email: email).exists?
    user = User::User.where(email: email).first
    unless user.password_is_right? password
      errors.add(:email, 'Wrong email or password')
    end
  else
    errors.add(:email, 'Wrong email or password')
  end
end
