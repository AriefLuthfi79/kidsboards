def initialize(middleware_list)
  last_middleware = nil
  middleware = middleware_list.pop
  while middleware
    middleware.next_middleware = last_middleware
    last_middleware = middleware
    middleware = middleware_list.pop
  end
  @middleware = last_middleware
end
